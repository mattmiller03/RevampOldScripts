var output;
var session;
var pollInterval = 10; // seconds between progress checks
var logFile = ScriptPath.replace(/\.ps1$/i, "_progress.log");

// Pull vCenter credentials from Aria config element (Assets > Configurations)
var category = Server.getConfigurationElementCategoryWithPath("DLAconfigs");
var vcUser = "";
var vcPass = "";

for each (var configuration in category.configurationElements) {
    if (configuration.name == "vCenter Creds") {
        vcUser = configuration.getAttributeWithKey("username").value;
        vcPass = configuration.getAttributeWithKey("password").value;
    }
}

if (!vcUser || !vcPass) {
    throw "Could not retrieve vCenter credentials from DLAconfigs/vCenter Creds";
}

// Username is passed as-is; the PS script appends SSODomain per vCenter from config JSON
System.log("Using vCenter credential base username: " + vcUser);

try {
    session = host.openSession();

    var escapedLog = logFile.replace(/\\/g, "\\\\");
    var escapedScript = ScriptPath.replace(/\\/g, "\\\\");

    // Helper: run a PS command via session and return the host output as a string
    function psExec(cmd) {
        var result = session.invokeScript(cmd);
        if (result.getInvocationState() == "Failed") {
            throw "PS error: " + result.getErrors();
        }
        var hostOut = result.getHostOutput();
        return hostOut ? hostOut.trim() : "";
    }

    // Build a PSCredential on the remote host and store in a variable for the session
    // The password is passed as a script parameter to avoid it appearing in command history
    var credCmd = '$secPass = ConvertTo-SecureString -String "' + vcPass.replace(/"/g, '`"') + '" -AsPlainText -Force; '
        + '$global:AriaCred = New-Object System.Management.Automation.PSCredential("' + vcUser + '", $secPass); '
        + 'Write-Host "Credential created"';

    psExec(credCmd);
    System.log("Credential prepared on remote host.");

    // Start the script as a background process via pwsh.exe, passing the credential via a temp file
    // Write the credential to a temp clixml so the child pwsh.exe process can read it
    var credFile = ScriptPath.replace(/\.ps1$/i, "_tempcred.xml");
    var escapedCredFile = credFile.replace(/\\/g, "\\\\");

    var saveCredCmd = '$global:AriaCred | Export-Clixml -Path "' + escapedCredFile + '" -Force; '
        + 'Write-Host "Credential saved to temp file"';
    psExec(saveCredCmd);

    // Build the pwsh.exe command that loads the cred and passes it to the script
    var psCmd = '$cred = Import-Clixml -Path \\\"' + escapedCredFile + '\\\"; '
        + '& \\\"' + escapedScript + '\\\" -VCenterCredential $cred; '
        + 'Remove-Item -Path \\\"' + escapedCredFile + '\\\" -Force -ErrorAction SilentlyContinue';

    var startCmd = '$logFile = "' + escapedLog + '"; '
        + 'Remove-Item -Path $logFile -Force -ErrorAction SilentlyContinue; '
        + '$proc = Start-Process -FilePath "C:\\Program Files\\PowerShell\\7\\pwsh.exe" '
        + '-ArgumentList @("-NoProfile","-NonInteractive","-Command","' + psCmd + '") '
        + '-RedirectStandardOutput $logFile -PassThru -NoNewWindow; '
        + 'Write-Host $proc.Id';

    var pid = psExec(startCmd);
    System.log("Script started as PID " + pid + ", polling progress...");

    // Poll the log file for new output until the process exits
    var lastLineCount = 0;
    var running = true;

    while (running) {
        System.sleep(pollInterval * 1000);

        // Check if process is still running
        var checkCmd = 'if (Get-Process -Id ' + pid + ' -ErrorAction SilentlyContinue) { Write-Host "RUNNING" } else { Write-Host "DONE" }';
        var state = psExec(checkCmd);
        running = (state === "RUNNING");

        // Read any new lines from the log file
        var readCmd = '$f = "' + escapedLog + '"; '
            + 'if (Test-Path $f) { $lines = @(Get-Content $f); '
            + 'if ($lines.Count -gt ' + lastLineCount + ') { '
            + 'Write-Host ($lines[' + lastLineCount + '..($lines.Count - 1)] -join "`n") '
            + '} }';

        var newText = psExec(readCmd);

        if (newText.length > 0) {
            var newLines = newText.split("\n");
            lastLineCount += newLines.length;

            for (var i = 0; i < newLines.length; i++) {
                System.log(newLines[i]);
            }
        }
    }

    // Final read to catch any remaining output
    var finalCmd = '$f = "' + escapedLog + '"; '
        + 'if (Test-Path $f) { $lines = @(Get-Content $f); '
        + 'if ($lines.Count -gt ' + lastLineCount + ') { '
        + 'Write-Host ($lines[' + lastLineCount + '..($lines.Count - 1)] -join "`n") '
        + '} }';

    var finalText = psExec(finalCmd);
    if (finalText.length > 0) {
        var finalLines = finalText.split("\n");
        for (var j = 0; j < finalLines.length; j++) {
            System.log(finalLines[j]);
        }
    }

    System.log("Script execution completed.");

} finally {
    // Clean up temp credential file if it still exists
    try {
        if (session) {
            session.invokeScript('Remove-Item -Path "' + escapedCredFile + '" -Force -ErrorAction SilentlyContinue');
        }
    } catch(e) { /* ignore cleanup errors */ }

    if (session) {
        host.closeSession(session.getSessionId());
    }
}
